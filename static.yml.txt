name: Deploy static site to Pages

on:
  # runs on every push to main
  push:
    branches: ['main']
  # ...and lets you trigger it by hand from the Actions tab
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

# allow one deployment at a time; don't cancel a run in progress
concurrency:
  group: 'pages'
  cancel-in-progress: false

jobs:
  deploy:
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Pages
        uses: actions/configure-pages@v5

      - name: Upload the whole repo as the site
        uses: actions/upload-pages-artifact@v3
        with:
          path: '.'

      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4
